### 3.3 `directionJobs` — 职业变体（delta 偏移）

**数量**: 6 组，每组 9 个，共 54 个职业。

```json
{
  "directionJobs": {
    "<directionId>": [
      {
        "id": "string — kebab-case 唯一标识",
        "name": "string — 中文职业名",
        "directionId": "string — 所属方向 ID",
        "icon": "string — 单个 emoji",
        "description": "string — ≤60 字，职业简述",
        "delta": {
          "nAch": 0.00,
          "nPow": 0.00,
          "nAff": 0.00,
          "TF": 0.05,
          "GM": 0.00,
          "AU": 0.05,
          "SE": 0.00,
          "EC": 0.00,
          "SV": 0.00,
          "CH": 0.00,
          "LS": 0.00,
          "wholistAnalytic": 0.00,
          "presence": 0.00,
          "search": 0.00
        }
      }
      // ... ×9
    ]
  }
}
```

**delta 约束**:
- 每个 delta 值 ∈ [-0.20, +0.20]
- 每个职业只修改 2-4 个维度（其余 delta=0.00）
- delta 字段必须包含全部 14 维（缺省填 0.00）
- `direction` 和 `weight` 从方向原型继承（职业不覆盖）

**delta 合并规则**（`resolveJobProfile` 实现）:

```js
function resolveJobProfile(directionProfile, jobVariant) {
  return {
    ...directionProfile,
    id: jobVariant.id,
    name: jobVariant.name,
    icon: jobVariant.icon,
    description: jobVariant.description,
    profiles: directionProfile.profiles.map(pf => {
      const delta = jobVariant.delta[pf.field] || 0;
      return {
        ...pf,
        ideal: Math.max(0, Math.min(1, pf.ideal + delta))
      };
    })
  };
}
```

### 5.2 delta 范围参照

| 调整程度 | delta 范围 | 含义 | 使用场景 |
|:---:|:---:|------|------|
| 微调 | ±0.05 | 小幅偏向 | 同方向内相似职业的细微差异 |
| 中调 | ±0.10 | 明显差异 | 同方向内职能明显不同的职业 |
| 大调 | ±0.15 ~ 0.20 | 显著变化 | 跨方向混合型职业（慎用） |

### 5.3 方向区分度验证标准

| 验证项 | 标准 | 验证方法 |
|------|------|---------|
| 方向剖面互异性 | 任意两个方向的 14 维剖面有明显差异 | 目视 + 欧氏距离 > 0.3 |
| Top1 方向可靠性 | 合成用户（极端型）匹配到预期方向 | T04 自动化测试 |
| 职业内区分度 | 同方向 9 个职业得分不全部相同 | 排名分布检查 (max-min > 5) |

### 5.4 数据质量检查规则

```js
// 每个方向需通过以下检查（matcher_extended.js 内置）
function validateDirectionProfile(dp) {
  if (dp.profiles.length !== 14) return 'profiles 长度必须为 14';
  if (!dp.id || !dp.name || !dp.slogan || !dp.icon) return '元数据不完整';
  const dimSet = new Set(dp.profiles.map(p => p.field));
  if (dimSet.size !== 14) return '维度重复或缺失';
  if (dp.gameWeights.G1+dp.gameWeights.G2+dp.gameWeights.G3+dp.gameWeights.G4 !== 1.0) return 'gameWeights 不求和为1';
  for (const pf of dp.profiles) {
    if (pf.ideal < 0 || pf.ideal > 1) return `ideal 值越界: ${pf.field}=${pf.ideal}`;
    if (!['high','mid','low'].includes(pf.direction)) return `invalid direction: ${pf.field}=${pf.direction}`;
    if (pf.weight <= 0 || pf.weight > 1) return `weight 越界: ${pf.field}=${pf.weight}`;
  }
  return null; // 通过
}
```
